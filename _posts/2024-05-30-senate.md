<script>
    function updateHeatmap() {
        var province = document.getElementById('province').value;
        var heatmapImage = document.getElementById('heatmap');
        heatmapImage.src = `https://raw.githubusercontent.com/naponjatusripitak/naponjatusripitak.github.io/master/assets/img/heatmaps/heatmap_${province}.jpg`;
    }

    document.addEventListener('DOMContentLoaded', () => {
        const heatmap = document.getElementById('heatmap');
        const container = document.querySelector('.heatmap-container');
        let scale = 1;
        let isZoomed = false;

        heatmap.addEventListener('click', () => {
            if (!isZoomed) {
                scale = 2; // Adjust this value for more zoom
                heatmap.style.transform = `scale(${scale})`;
                heatmap.style.cursor = 'zoom-out';
            } else {
                scale = 1;
                heatmap.style.transform = `scale(${scale})`;
                heatmap.style.cursor = 'zoom-in';
            }
            isZoomed = !isZoomed;
        });

        container.addEventListener('wheel', (event) => {
            event.preventDefault();
            const delta = event.deltaY;
            if (delta < 0) {
                scale += 0.1;
            } else {
                scale -= 0.1;
            }
            if (scale < 1) scale = 1;
            if (scale > 3) scale = 3; // Max zoom level
            heatmap.style.transform = `scale(${scale})`;
        });

        container.addEventListener('touchstart', handleTouchStart, false);
        container.addEventListener('touchmove', handleTouchMove, false);

        let x1 = null;
        let y1 = null;

        function handleTouchStart(event) {
            const firstTouch = event.touches[0];
            x1 = firstTouch.clientX;
            y1 = firstTouch.clientY;
        }

        function handleTouchMove(event) {
            if (!x1 || !y1) {
                return;
            }

            let x2 = event.touches[0].clientX;
            let y2 = event.touches[0].clientY;

            let xDiff = x2 - x1;
            let yDiff = y2 - y1;

            container.scrollLeft -= xDiff;
            container.scrollTop -= yDiff;

            x1 = x2;
            y1 = y2;
        }
    });
</script>
